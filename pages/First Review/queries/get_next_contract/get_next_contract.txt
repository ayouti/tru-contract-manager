-- The approval_date >= '2026-06-01' filter is intentional — it is the go-live date and excludes pre-migration records for performance.

UPDATE production.received_contracts
SET 
  first_reviewed_by = {{appsmith.user.email}},
  locked_at = NOW(),
  status = CASE 
    WHEN status LIKE '%RECEIVED%' THEN 'IN_PROGRESS_1ST_RECEIVED'
    ELSE 'IN_PROGRESS_1ST'
  END
WHERE loan_application_id = (
  SELECT loan_application_id 
  FROM production.received_contracts
  WHERE first_review_date IS NULL
    AND second_review_date IS NULL
    AND (
      -- Condition 1: It's brand new
      status IN ('PENDING_1ST_REVIEW', 'RECEIVED') OR status IS NULL
      -- Condition 2: It was locked, but the lock expired (catches IN_PROGRESS ghosts)
      OR locked_at < NOW() - INTERVAL '10 minutes'
    ) 
			AND approval_date >= '2026-06-01'
  ORDER BY
	
	approval_date ASC 
  LIMIT 1
  FOR UPDATE SKIP LOCKED
)
RETURNING *;
