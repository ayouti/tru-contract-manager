INSERT INTO production.received_contracts (
    loan_application_id, approval_date, agent_id, agent_name, branch, 
    customer_name, customer_phone, customer_national_id, approval_limit, 
    received_by, shipment_received_date, contract_received_date, shipment_finish_date,
    shipment_number, drawer_number, comment, contract_type, status
)
SELECT 
    (elem->>'loan_application_id')::INTEGER,
    (NULLIF(elem->>'approval_date', ''))::TIMESTAMP,
    (NULLIF(elem->>'agent_id', ''))::INTEGER,
    elem->>'agent_name',
    elem->>'branch',
    elem->>'customer_name',
    elem->>'customer_phone',
    elem->>'customer_national_id',
    (NULLIF(elem->>'approval_limit', ''))::NUMERIC,
    elem->>'received_by',
    (NULLIF(elem->>'shipment_received_date', ''))::TIMESTAMP,
    (NULLIF(elem->>'contract_received_date', ''))::TIMESTAMP,
    (NULLIF(elem->>'shipment_finish_date', ''))::TIMESTAMP,
    elem->>'shipment_number',
    (NULLIF(elem->>'drawer_number', ''))::INTEGER,
    elem->>'comment',
    COALESCE(NULLIF(elem->>'contract_type', ''), 'HANDWRITTEN'),
    'RECEIVED'
FROM json_array_elements({{ this.params.rows }}::json) AS elem
ON CONFLICT (loan_application_id) 
DO UPDATE SET 
    received_by = EXCLUDED.received_by,
    shipment_received_date = EXCLUDED.shipment_received_date,
    contract_received_date = EXCLUDED.contract_received_date,
    shipment_finish_date = EXCLUDED.shipment_finish_date,
    shipment_number = EXCLUDED.shipment_number,
    drawer_number = EXCLUDED.drawer_number,
    comment = EXCLUDED.comment,
    contract_type = EXCLUDED.contract_type,
    status = CASE
        WHEN production.received_contracts.status = 'IN_PROGRESS_1ST' THEN 'IN_PROGRESS_1ST_RECEIVED'
        WHEN production.received_contracts.status = 'IN_PROGRESS_1ST_RECEIVED' THEN 'IN_PROGRESS_1ST_RECEIVED'
        ELSE 'RECEIVED'
    END;
